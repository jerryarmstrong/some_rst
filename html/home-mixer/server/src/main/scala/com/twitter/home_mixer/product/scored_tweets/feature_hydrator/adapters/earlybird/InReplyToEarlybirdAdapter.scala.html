)

InReplyToTweetTimelinesSharedFeatures.WEIGHTED_RETWEET_COUNT, features.weightedRetweetCount.map(_.toDouble)

)

InReplyToTweetTimelinesSharedFeatures.WEIGHTED_REPLY_COUNT, features.weightedReplyCount.map(_.toDouble)

)

InReplyToTweetTimelinesSharedFeatures.WEIGHTED_QUOTE_COUNT, features.weightedQuoteCount.map(_.toDouble)

)

richDataRecord.setFeatureValueFromOption(

InReplyToTweetTimelinesSharedFeatures.DECAYED_FAVORITE_COUNT, features.decayedFavoriteCount.map(_.toDouble)

)

richDataRecord.setFeatureValueFromOption(

InReplyToTweetTimelinesSharedFeatures.DECAYED_RETWEET_COUNT, features.decayedRetweetCount.map(_.toDouble)

)

InReplyToTweetTimelinesSharedFeatures.DECAYED_REPLY_COUNT, features.decayedReplyCount.map(_.toDouble)

)

InReplyToTweetTimelinesSharedFeatures.QUOTE_COUNT, features.quoteCount.map(_.toDouble)

)

InReplyToTweetTimelinesSharedFeatures.EARLYBIRD_SCORE, features.earlybirdScore

)

InReplyToRecapFeatures.PREV_USER_TWEET_ENGAGEMENT, features.prevUserTweetEngagement.toDouble

)

.setFeatureValue[JBoolean](InReplyToRecapFeatures.IS_SENSITIVE, features.isSensitiveContent)

richDataRecord

.setFeatureValue[JBoolean](InReplyToRecapFeatures.IS_AUTHOR_NEW, features.isAuthorNew)

richDataRecord.setFeatureValue[JDouble](

InReplyToRecapFeatures.NUM_MENTIONS, features.numMentions.toDouble)

richDataRecord

.setFeatureValue[JBoolean](InReplyToRecapFeatures.HAS_MENTION, (features.numMentions > 0))

richDataRecord

.setFeatureValue[JBoolean](InReplyToRecapFeatures.HAS_HASHTAG, (features.numHashtags > 0))

richDataRecord

.setFeatureValue[JBoolean](InReplyToRecapFeatures.IS_AUTHOR_NSFW, features.isAuthorNSFW)

.setFeatureValue[JBoolean](InReplyToRecapFeatures.IS_AUTHOR_SPAM, features.isAuthorSpam)

.setFeatureValue[JBoolean](InReplyToRecapFeatures.IS_AUTHOR_BOT, features.isAuthorBot)

InReplyToRecapFeatures.FROM_MUTUAL_FOLLOW, features.fromMutualFollow)

InReplyToRecapFeatures.FROM_VERIFIED_ACCOUNT, features.fromVerifiedAccount)

richDataRecord.setFeatureValue[JDouble](

InReplyToRecapFeatures.UNIDIRECTIONAL_FAV_COUNT, features.unidirectionalFavCount)

richDataRecord.setFeatureValue[JDouble](

InReplyToRecapFeatures.CONVERSATIONAL_COUNT, features.conversationCount)

richDataRecord

.setFeatureValue[JDouble](InReplyToRecapFeatures.REPLY_COUNT, features.replyCount.toDouble)

InReplyToRecapFeatures.RETWEET_COUNT, features.retweetCount.toDouble)

.setFeatureValue[JDouble](InReplyToRecapFeatures.FAV_COUNT, features.favCount.toDouble)

InReplyToRecapFeatures.RETWEET_COUNT_V2, features.retweetCountV2.map(_.toDouble)

) richDataRecord.setFeatureValueFromOption(

InReplyToRecapFeatures.REPLY_COUNT_V2, features.replyCountV2.map(_.toDouble))

}

}
