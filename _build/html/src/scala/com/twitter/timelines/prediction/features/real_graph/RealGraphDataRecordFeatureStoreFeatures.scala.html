}

)

def convertFeatureToLegacyName(

prefix: String, variance: String = “variance”

): Map[Feature[EdgeEntityId[UserId, UserId], _ >: Long with Double <: AnyVal], String] =
edgeFeatureToLegacyName.flatMap {
case (k, v) =>
Seq(

k.NonZeroDays -> s”${prefix}.${v}.non_zero_days”, k.DaysSinceLast -> s”${prefix}.${v}.days_since_last”, k.ElapsedDays -> s”${prefix}.${v}.elapsed_days”, k.Ewma -> s”${prefix}.${v}.ewma”, k.Mean -> s”${prefix}.${v}.mean”, k.Variance -> s”${prefix}.${v}.${variance}”,

)

} ++ Map(

RealGraph.Weight -> (prefix + “.weight”)

)

}
