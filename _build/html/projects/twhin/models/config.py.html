class TwhinEmbeddingsConfig(LargeEmbeddingsConfig):

@validator(“tables”) def embedding_dims_match(cls, tables):

embedding_dim = tables[0].embedding_dim data_type = tables[0].data_type for table in tables:

assert table.embedding_dim == embedding_dim, “Embedding dimensions for all nodes must match.” assert table.data_type == data_type, “Data types for all nodes must match.”

return tables
