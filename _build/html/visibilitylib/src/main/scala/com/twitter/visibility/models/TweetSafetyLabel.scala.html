thriftToModelMap.get(safetyLabelType) match {

case Some(tweetSafetyLabelType) => tweetSafetyLabelType case _ =>

safetyLabelType match {

case s.SafetyLabelType.EnumUnknownSafetyLabelType(DeprecatedEnumValue) => Deprecated case _ =>

Unknown

}

}

modelToThriftMap.getOrElse(safetyLabelType, UnknownThriftSafetyLabelType)

}
