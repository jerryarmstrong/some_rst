/**

  • Class that can serialize and deserialize an iterator of thrift objects.

  • This class can do things lazily so there is no need to have all the object into memory.

*/

codec: ThriftStructCodec[T]) { def toOutputStream(

iterator: Iterator[T], outputStream: OutputStream

): Unit = {

val protocol = (new TBinaryProtocol.Factory).getProtocol(new TIOStreamTransport(outputStream)) iterator.foreach { thriftObject =>

}

}

def fromInputStream(

ThriftIteratorIO.getIterator(codec, inputStream)

}

}
