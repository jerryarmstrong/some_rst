REGEX_PATTERNS = [

r’^RT @[A-Za-z0-9_]+: ‘, r”@[A-Za-z0-9_]+”, r’https://t.co/[A-Za-z0-9]{10}’, r’@?????’,

]

EMOJI_PATTERN = re.compile(

“([” “U0001F1E0-U0001F1FF” “U0001F300-U0001F5FF” “U0001F600-U0001F64F” “U0001F680-U0001F6FF” “U0001F700-U0001F77F” “U0001F780-U0001F7FF” “U0001F800-U0001F8FF” “U0001F900-U0001F9FF” “U0001FA00-U0001FA6F” “U0001FA70-U0001FAFF” “U00002702-U000027B0” “])”

)

def clean_tweet(text):
for pattern in REGEX_PATTERNS:

text = re.sub(pattern, ‘’, text)

text = re.sub(EMOJI_PATTERN, r’ 1 ‘, text)

text = re.sub(r’n’, ‘ ‘, text)

return text.strip().lower()
