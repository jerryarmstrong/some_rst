@Singleton class TombstoneDisplayTypeMarshaller @Inject() () {

def apply(tombstoneDisplayType: TombstoneDisplayType): urt.TombstoneDisplayType =
tombstoneDisplayType match {

case TweetUnavailable => urt.TombstoneDisplayType.TweetUnavailable case DisconnectedRepliesAncestor => urt.TombstoneDisplayType.DisconnectedRepliesAncestor case DisconnectedRepliesDescendant => urt.TombstoneDisplayType.DisconnectedRepliesDescendant case Inline => urt.TombstoneDisplayType.Inline case NonCompliant => urt.TombstoneDisplayType.NonCompliant

}

}
