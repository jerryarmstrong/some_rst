def requireSome[A, B](l: Lens[A, Option[B]]): Lens[A, B] =
checkEq[A, B](

a => l.get(a).get, (a, b) => l.set(a, Some(b))

)

def tweetLens[A](get: Tweet => A, set: (Tweet, A) => Tweet): Lens[Tweet, A] =

checkEq[Tweet, A](get, set)
