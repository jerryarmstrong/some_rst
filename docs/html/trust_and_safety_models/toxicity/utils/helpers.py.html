def compute_precision_fixed_recall(labels, preds, fixed_recall):

precision_values, recall_values, thresholds = precision_recall_curve(y_true=labels, probas_pred=preds) index_recall = bisect.bisect_left(-recall_values, -1 * fixed_recall) result = precision_values[index_recall - 1] print(f”Precision at {recall_values[index_recall-1]} recall: {result}”)

return result, thresholds[index_recall - 1]
